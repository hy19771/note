	function sct() {
		return document.documentElement.scrollTop || window.pageYOffset || document.body.scrollTop || 0;
	}

###获取元素的内联样式
	function getStyle(ele, styleName) {
	if (ele.currentStyle) {
		return ele.currentStyle[styleName];
	} else {
		return window.getComputedStyle(ele, null)[styleName];
		}
	}
